# 그리드 선택 문제 해결 가이드

## 🔍 문제 원인 분석

### 1. **함수 스코프 문제**
```javascript
function initParentSearchGrid() {
    // ❌ 문제: selectParent 함수가 initParentSearchGrid 함수 안에 있음
    function selectParent(data) {
        // 이 함수는 initParentSearchGrid 밖에서 접근 불가능!
    }
    
    // onCellClicked에서 selectParent를 호출하려고 하지만...
    onCellClicked: function(params) {
        selectParent(params.data); // ❌ 스코프 밖에서 호출 불가능
    }
}
```

**왜 문제인가?**
- `selectParent` 함수가 `initParentSearchGrid` 함수 **안에** 정의되어 있음
- ag-grid의 `onCellClicked` 이벤트는 나중에 실행되는데, 그때는 `selectParent` 함수에 접근할 수 없음
- JavaScript의 스코프 규칙 때문에 함수 안에서 정의된 함수는 밖에서 접근 불가능

### 2. **동적 버튼 생성 문제**
```javascript
cellRenderer: function(params) {
    // ❌ 문제: 버튼을 HTML 문자열로 생성
    return '<button class="select-btn">선택</button>';
}
```

**왜 문제인가?**
- `cellRenderer`는 HTML 문자열만 반환함
- 실제 버튼이 DOM에 추가되는 시점과 이벤트 리스너를 연결하는 시점이 다름
- ag-grid가 내부적으로 버튼을 생성/삭제하면서 이벤트가 끊어질 수 있음

### 3. **데이터 업데이트 후 이벤트 끊김**
```javascript
// 데이터를 업데이트하면
parentSearchGridApi.setGridOption('rowData', newData);

// ❌ 문제: 기존 버튼들이 삭제되고 새로 생성됨
// 하지만 이벤트 리스너는 다시 연결되지 않음!
```

## ✅ 해결 방법

### 방법 1: 함수를 전역 스코프로 이동 (가장 중요!)

```javascript
// ✅ 올바른 방법: 함수를 전역 스코프에 정의
function selectParentFromGrid(data) {
    if (!data) return;
    
    const userId = data.userId || '';
    const name = data.name || '';
    
    document.getElementById('modalParentName').value = name;
    document.getElementById('modalParentName').setAttribute('data-userid', userId);
    closeParentSearchModal();
}

function initParentSearchGrid() {
    // 이제 selectParentFromGrid를 어디서든 호출 가능!
    onCellClicked: function(params) {
        if (params.colDef.headerName === '선택') {
            selectParentFromGrid(params.data); // ✅ 작동함!
        }
    }
}
```

**왜 작동하는가?**
- 전역 스코프에 있으면 어디서든 접근 가능
- ag-grid 이벤트 핸들러에서도 호출 가능

### 방법 2: onCellClicked 사용 (ag-grid 권장 방법)

```javascript
const parentSearchColumnDefs = [
    { field: 'name', headerName: '성명', width: 120 },
    {
        headerName: '선택',
        width: 80,
        cellRenderer: function(params) {
            // 버튼 HTML만 반환
            return '<button class="select-btn">선택</button>';
        },
        // ✅ 핵심: onCellClicked에서 처리
        onCellClicked: function(params) {
            // params.data에 해당 행의 데이터가 들어있음
            if (params.data) {
                selectParentFromGrid(params.data); // ✅ 작동!
            }
        }
    }
];
```

**왜 작동하는가?**
- ag-grid가 셀 클릭을 감지하고 `onCellClicked`를 호출
- `params.data`에 해당 행의 전체 데이터가 들어있음
- 버튼을 클릭해도 셀 클릭으로 인식됨

### 방법 3: 데이터 업데이트 후 이벤트 재연결

```javascript
// 데이터를 업데이트한 후
parentSearchGridApi.setGridOption('rowData', data);

// ✅ 버튼 이벤트를 다시 연결
setTimeout(function() {
    const gridElement = document.querySelector('#parentSearchGrid');
    const selectButtons = gridElement.querySelectorAll('.select-btn');
    
    selectButtons.forEach(function(btn, index) {
        // 기존 이벤트 제거 후 새로 추가
        const newBtn = btn.cloneNode(true);
        btn.parentNode.replaceChild(newBtn, btn);
        
        newBtn.addEventListener('click', function(e) {
            e.stopPropagation(); // 이벤트 버블링 방지
            
            // 행 인덱스로 데이터 가져오기
            const rowNode = parentSearchGridApi.getRowNode({ rowIndex: index });
            if (rowNode && rowNode.data) {
                selectParentFromGrid(rowNode.data);
            }
        });
    });
}, 200); // 그리드 렌더링 완료 대기
```

## 📝 완전한 예제 코드

```javascript
// 1. 전역 스코프에 선택 함수 정의
function selectParentFromGrid(data) {
    if (!data || !data.userId) {
        alert('사용자 정보가 없습니다.');
        return;
    }
    
    document.getElementById('modalParentName').value = data.name;
    document.getElementById('modalParentName').setAttribute('data-userid', data.userId);
    closeParentSearchModal();
}

// 2. 그리드 초기화
function initParentSearchGrid() {
    const gridElement = document.querySelector('#parentSearchGrid');
    if (!gridElement) return;
    
    // 기존 그리드 제거
    if (parentSearchGridApi) {
        parentSearchGridApi.destroy();
    }
    
    // 컬럼 정의
    const parentSearchColumnDefs = [
        { field: 'name', headerName: '성명', width: 120, sortable: true },
        { field: 'phone', headerName: '전화번호', width: 150, sortable: true },
        { field: 'user_type_nm', headerName: '사용자구분', width: 120, sortable: true },
        {
            headerName: '선택',
            width: 80,
            cellRenderer: function(params) {
                return '<button class="select-btn">선택</button>';
            },
            // ✅ 핵심: 셀 클릭 이벤트
            onCellClicked: function(params) {
                if (params.data) {
                    selectParentFromGrid(params.data);
                }
            }
        }
    ];
    
    // 그리드 옵션
    const parentSearchGridOptions = {
        columnDefs: parentSearchColumnDefs,
        rowSelection: 'single',
        suppressRowClickSelection: false,
        // 더블클릭으로도 선택 가능
        onRowDoubleClicked: function(event) {
            if (event.data) {
                selectParentFromGrid(event.data);
            }
        },
        defaultColDef: {
            resizable: true
        },
        domLayout: 'normal',
        rowHeight: 40
    };
    
    // 그리드 생성
    parentSearchGridApi = agGrid.createGrid(gridElement, parentSearchGridOptions);
}

// 3. 데이터 로드 후 처리
function searchParentUsers() {
    // ... 데이터 조회 로직 ...
    
    .then(function(data) {
        if (parentSearchGridApi) {
            // 데이터 설정
            parentSearchGridApi.setGridOption('rowData', data);
            
            // ✅ 데이터 업데이트 후 버튼 이벤트 재연결 (선택사항)
            // onCellClicked를 사용하면 자동으로 작동하지만,
            // 추가 안전장치로 직접 이벤트 연결도 가능
        }
    });
}
```

## 🎯 핵심 포인트 정리

1. **함수는 전역 스코프에 정의** - ag-grid 이벤트에서 접근 가능하도록
2. **onCellClicked 사용** - ag-grid가 제공하는 공식 방법
3. **params.data 활용** - 행의 전체 데이터를 바로 사용
4. **이벤트 버블링 주의** - `e.stopPropagation()` 사용

## ❓ 자주 묻는 질문

**Q: 왜 버튼을 클릭해도 아무 일도 안 일어나나요?**
A: 함수가 스코프 밖에 있거나, `onCellClicked`가 제대로 설정되지 않았을 수 있습니다.

**Q: 더블클릭은 되는데 버튼 클릭이 안 되요**
A: `onCellClicked`가 제대로 설정되었는지 확인하세요. 버튼 클릭도 셀 클릭으로 인식되어야 합니다.

**Q: 데이터를 다시 로드하면 버튼이 작동 안 해요**
A: `onCellClicked`를 사용하면 자동으로 작동합니다. 직접 이벤트 리스너를 연결했다면 재연결이 필요합니다.
